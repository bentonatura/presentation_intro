\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{arrows, automata, positioning, quotes}

\begin{document}
\begin{tikzpicture}[auto]

\def \n {5}
\def \radius {3cm}
\def \margin {3} % margin in angles, depends on the radius

\foreach \s in {1,...,\n}
{
  \pgfmathsetmacro\num{int(2 * (\s - 2 * int(\s/2))- 1)};
  \node[draw, circle] at ({360/\n * (\s - 1)}:\radius) {$ $};
  \draw[-, >=latex,label=] ({360/\n * (\s - 1)+\margin}:\radius) 
    arc ({360/\n * (\s - 1)+\margin}:{360/\n * (\s)-\margin}:\radius) node[midway]{$\num$};
}
\end{tikzpicture}
\end{document}
